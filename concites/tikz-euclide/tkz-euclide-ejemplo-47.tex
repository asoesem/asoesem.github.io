\begin{document}
    \begin{tikzpicture}
        % Define el centro C de la elipse.
        \tkzDefPoint(0,4){C}
    
        % Dibuja una elipse centrada en C con semiejes 4 y 2, rotada 45Â°.
        \tkzDrawEllipse[blue](C,4,2,45)
    
        % Dibuja y etiqueta el centro.
        \tkzDrawPoints(C)
        \tkzLabelPoints(C)
    \end{tikzpicture}
