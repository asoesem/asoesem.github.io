\documentclass[tikz,border=2mm]{standalone}
\usepackage{tkz-base}
\usepackage{tkz-euclide}

\begin{document}
    \begin{tikzpicture}
        % Define el sistema de coordenadas ortogonales.
        \tkzInit[xmin=-3,xmax=3,ymin=-3,ymax=3]
        
        % Dibuja los ejes x, y
        \tkzDrawXY

        % Etiqueta los ejex x, y
        \tkzLabelXY
    \end{tikzpicture}
\end{document}
